  const em=document.getElementById('lEmail'),ps=document.getElementById('lPass');
  clearErr('login');

function loadComments(){
  if(allLoaded)return;
  const list=document.getElementById('commentsList');
  const s=cPage*CPP,en=Math.min(s+CPP,comments.length);
  if(s>=comments.length){allLoaded=true;document.getElementById('loadMore').style.display='none';return}
  for(let i=s;i<en;i++){
    const c=comments[i],div=document.createElement('div');
    div.className='c-item';div.setAttribute('role','listitem');
    div.innerHTML=`
      <div class="c-av"><img src="${c.avatar}" alt="${c.name}"></div>
      <div class="c-body">
        <div class="c-name">${c.name}</div>
        <div class="c-text">${c.text}</div>
        <div class="c-meta">
          <span>${c.time}</span>
          <button onclick="likeC(this,${c.likes})">Suka</button>
          <button onclick="replyC('${c.name}')">Balas</button>
        </div>
      </div>`;
    list.appendChild(div);
  }
  cPage++;
  if(en>=comments.length){allLoaded=true;document.getElementById('loadMore').style.display='none'}
}

function likeC(btn,n){
  if(btn.classList.contains('liked'))return;
  btn.classList.add('liked');btn.style.color='var(--red)';btn.style.fontWeight='600';
  btn.textContent='Suka ('+String(n+1)+')';
}
function replyC(name){
  if(!loggedIn){openModal('login');return}
  const i=document.getElementById('commentInput');
  i.value='@'+name+' ';i.focus();
}

function submitComment(){
  const i=document.getElementById('commentInput'),t=i.value.trim();
  if(!t)return;
  if(!loggedIn){openModal('login');return}
  const list=document.getElementById('commentsList'),div=document.createElement('div');
  div.className='c-item';div.style.animation='ddIn .3s ease';
  div.innerHTML=`
    <div class="c-av"><img src="<?php echo $banner; ?>" alt="Anda"></div>
    <div class="c-body">
      <div class="c-name">Member <?php echo strtoupper($titlebrand); ?></div>
      <div class="c-text">${escHtml(t)}</div>
      <div class="c-meta">
        <span>Baru saja</span>
        <button onclick="likeC(this,0)">Suka</button>
        <button onclick="replyC('Member <?php echo strtoupper($titlebrand); ?>')">Balas</button>
      </div>
    </div>`;
  list.insertBefore(div,list.firstChild);i.value='';
  const ce=document.getElementById('commentCount'),cur=parseInt(ce.textContent)||0;
  ce.textContent=String(cur+1);showToast('Komentar terkirim');
}

function escHtml(s){const d=document.createElement('div');d.textContent=s;return d.innerHTML}

function setupTA(){
  const ta=document.getElementById('commentInput');

  ta.addEventListener('keydown',e=>{if(e.key==='Enter'&&!e.shiftKey){e.preventDefault();submitComment()}});
}
